bars = plt.bar(range(len(seed_avg_volume_sorted)), seed_avg_volume_sorted.values, 
               color=plt.cm.Set3(np.linspace(0, 1, len(seed_avg_volume_sorted))))
plt.title('Average Search Volume by Seed Keyword (Sorted by Volume - High to Low)', fontsize=16, fontweight='bold')
plt.xlabel('Seed Keywords')
plt.ylabel('Average Search Volume')

# Set Y-axis to start from 0 and add grid for better visibility
plt.ylim(0, max(seed_avg_volume_sorted.values) * 1.1)
plt.grid(axis='y', alpha=0.3, linestyle='--')

plt.xticks(range(len(seed_avg_volume_sorted)), seed_avg_volume_sorted.index, rotation=45, ha='right')
for bar, volume in zip(bars, seed_avg_volume_sorted.values):
    plt.text(bar.get_x() + bar.get_width()/2, bar.get_height() + max(seed_avg_volume_sorted.values) * 0.02, 
             f'{volume:.0f}', ha='center', va='bottom', fontweight='bold')
plt.tight_layout()
plt.show()
